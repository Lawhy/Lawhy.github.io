The Gym-Like Interface

Each step() is a full agent episode — the environment author specifies what's unique to the task.

Contract

class Environment:
    async def reset(self):
        """Async resource initialization."""
    async def step(self, action) -> StepResult:
        """Run one full agent episode."""
    async def cleanup(self):
        """Async resource teardown."""
    def get_tools(self) -> list[Tool]:
        """Tools available to the agent."""
    def get_hooks(self) -> list[HookProvider]:
        """Additional agent hooks."""

Example

# Docker environment example
async def reset(self):
    self.docker_env = DockerEnvironment(...)
    await self.docker_env.init()
async def step(self, action):
    agent = Agent(model, tools, hooks)
    error = None
    try:
        # run the agent loop
        await agent.invoke_async(action.message)
    except Exception as e:
        error = e
    termination = TerminationReason.from_error(error)
    observation = build_observation(agent)
    reward = await self.reward_fn.compute(...)
    return StepResult(observation, reward, termination)
# Docker environment example
async def cleanup(self):
    await self.docker_env.stop()
# Web search environment example
@tool
def search(query: str, topk: int):
    """Search the web for a query."""

@tool
def scrape(url: str, summary_instruction: str):
    """Scrape a URL and summarize content."""

def get_tools(self):
    return [search, scrape]
def get_hooks(self):
    return [
        ToolLimiter(
            max_tool_iters=10,
            max_tool_calls=50,
            max_parallel_tool_calls=5,
        )
    ]

Environments are per-rollout — each prompt maps to a different task, so each rollout gets its own environment instance with fresh state.
